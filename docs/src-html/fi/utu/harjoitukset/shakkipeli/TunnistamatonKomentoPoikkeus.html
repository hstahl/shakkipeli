001package fi.utu.harjoitukset.shakkipeli;
002/**
003 * Poikkeustyyppi, joka on tarkoitus heittää ohjelman
004 * lukiessa komentoparametriä konsolistasyötteestä ja
005 * saatuaan väärin muotoillun tai muuten virheellisen
006 * komennon.
007 */
008public class TunnistamatonKomentoPoikkeus extends Exception {
009
010        private String komento;
011        
012        /**
013         * Luo <code>TunnistamatonKomentoPoikkeus</code>:ksen ilman kuvaavaa viestiä.
014         */
015        public TunnistamatonKomentoPoikkeus() {}
016
017        /**
018         * Luo <code>TunnistamatonKomentoPoikkeus</code>:ksen kuvaavan viestin kanssa.
019         *
020         * @param komento välitettävä virheviesti
021         */
022        public TunnistamatonKomentoPoikkeus(String komento) {
023                super(komento);
024                this.komento = komento;
025        }
026
027        /**
028         * Poikkeusta kuvaava merkkijonoesitys
029         * 
030         * @return poikkeuksen merkkijonoesitys
031         */
032        public String toString() {
033                return "Tunnistamaton komento: " + komento;
034        }
035}

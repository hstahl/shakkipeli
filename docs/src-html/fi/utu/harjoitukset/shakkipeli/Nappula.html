001package fi.utu.harjoitukset.shakkipeli;
002/**
003 * Shakkipelin nappuloiden yhteinen rajapinta.
004 * Oletettavasti tällä tulee olemaan kuusi
005 * konkreettia perivää luokkaa, joista kukin
006 * kuvaa yhdentyyppistä shakkinappulaa.
007 * <p>
008 * Nappulat ovat joko mustia tai valkoisia ja
009 * niille on sallittu jotkin siirrot vain, jos
010 * niitä ei ole vielä siirretty pelissä.
011 * Näiden ominaisuuksien havainnointia varten
012 * määritellään metodit tässä rajapinnassa.
013 * Nappulan <code>toString()</code> -esityksen
014 * on tarkoitus kertoa jollain tavalla,
015 * yhdellä merkillä, sen tyyppi ja väri.
016 * <code>onLaillinen()</code> kertoo onko
017 * sille parametreillä ehdotettu siirto
018 * sallittu shakin säännöissä. Kukin perivä
019 * nappulatyyppi toteuttaa tämän metodin
020 * omalla tavallaan.
021 */
022public interface Nappula {
023
024        /**
025         * Literaaliluokka, jonka on tarkoitus
026         * määrittää eri nappulatyyppien
027         * merkkiesitys shakkilaudan
028         * konsolitulosteessa.
029         * <p>
030         * Huom. <code>Shakkipeli</code> -luokassa
031         * käytetään parametreinä useassa metodissa
032         * ja pelaajilta pyydetyssä syötteessä
033         * omia merkkiesityksiä, jotka eivät
034         * välttämättä ole samoja kuin tässä.
035         * Tämän literaaliluokan on tarkoitus
036         * määrittää vain konsolille piirrettävän
037         * pelilautanäkymän merkkiesitykset,
038         * eli nappuloiden <code>toString()</code>
039         * käyttäytyminen.
040         * <p>
041         * Koska ympäröivä rajapinta kuvaa
042         * shakkinappulan käyttäytymistä pelissä,
043         * oletettavasti erilaisia nappuloita on,
044         * väriä huomioimatta, tasan kuusi
045         * kappaletta ja niiden nimet ovat tämän
046         * literaaliluokan vakiot.
047         */
048        public enum MerkkiesitysNappulalle {
049                SOTILAS ("S", "s"),
050                LAHETTI ("L", "l"),
051                RATSU ("R", "r"),
052                TORNI ("T", "t"),
053                KUNINGATAR ("D", "d"),
054                KUNINGAS ("K", "k");
055                
056                private final String VALK;
057                private final String MUST;
058                
059                MerkkiesitysNappulalle(String VALK, String MUST) {
060                        this.VALK = VALK;
061                        this.MUST = MUST;
062                }
063                
064                /**
065                 * Antaa merkkiesityksen tämäntyyppiselle
066                 * valkoiselle nappulalle.
067                 * 
068                 * @return merkkiesitys tällaiselle valkoiselle nappulalle
069                 */
070                public String valkoinen() {
071                        return VALK;
072                }
073                
074                /**
075                 * Antaa merkkiesityksen tämäntyyppiselle
076                 * mustalle nappulalle.
077                 *
078                 * @return merkkiesitys tällaiselle mustalle nappulalle
079                 */             
080                public String musta() {
081                        return MUST;
082                }
083        }
084
085        /**
086         * Kertoo, onko nappula valkoinen
087         *
088         * @return <code>true</code>, jos nappula on valkoisen pelaajan.
089         */
090        public boolean onValkoinen();
091        
092        /** 
093         * Havannoi, onko nappulaa sirretty. Joitakin siirtoja
094         * voi tehdä vain ensimmäisellä kerralla siirrettäessä
095         * nappulaa.
096         * 
097         * @return <code>true</code>, jos nappulaa on sirretty
098         */
099        public boolean onSiirretty();
100        
101        /**
102         * Totuusarvometodi nappulalle sallittujen siirtojen
103         * tarkastamista varten. Palauttaa <code>true</code>,
104         * jos shakin säännöt sallivat siirron tällä
105         * nappulalla.
106         *
107         * @return <code>true</code>, jos ehdotettu siirto on sallittu tälle nappulalle
108         * @param peli <code>Shakkipeli</code>, jossa siirto halutaan tehdä
109         * @param vaakaAlku lähtökohdan vaakakoordinaatti [A-Ha-h]
110         * @param pystyAlku lähtökohdan pystykoordinaatti [1-8]
111         * @param vaakaLoppu maaliruudun vaakakoordinaatti [A-Ha-h]
112         * @param pystyLoppu maaliruudun pystykoordinaatti [1-8]
113         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaAlku == c) && (1 <= pystyAlku <= 8) && EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaLoppu == c) && (1 <= pystyLoppu <= 8)</code>
114         */
115        public boolean onLaillinen(Shakkipeli peli, char vaakaAlku, int pystyAlku,
116                                                                                                char vaakaLoppu, int pystyLoppu);
117        
118        /**
119         * Shakkinappulan merkkijonoesitys.
120         * <p>
121         * Esityksen tulee olla jokin <code>MerkkiesitysNappulalle</code>
122         * literaaliluokan vakioista siten, että nappulan
123         * <code>onValkoinen()</code> ollessa tosi, käytetään
124         * merkkiesitystä, jonka antaa <code>MerkkiesitysNappulalle</code>
125         * kutakin nappulatyyppiä vastaavan vakion
126         * <code>valkoinen()</code> -metodi, ja toisessa
127         * tapauksessa <code>musta()</code> -metodi.
128         *
129         * @return <code>Nappula</code>:n esitys String:nä
130         */
131        public String toString();
132}

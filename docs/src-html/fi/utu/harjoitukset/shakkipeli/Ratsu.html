001package fi.utu.harjoitukset.shakkipeli;
002import java.lang.Math;
003/**
004 * Ratsu- shakkinappula
005 */
006public class Ratsu extends AbstraktiNappula {
007        
008        /**
009         * Uutta ratsua luodessa kutsuttava konstruktori
010         *
011         * @param onValkoinen Halutaanko luoda valkoinen nappula?
012         * @.post <code>this.onValkoinen() == onValkoinen</code>
013         */
014        public Ratsu(boolean onValkoinen) {
015                super(MerkkiesitysNappulalle.RATSU, onValkoinen);
016        }
017
018        /**
019         * Ratsua voi siirtää ensin kaksi ruutua vaaka- tai pystyakselin
020         * suunnassa ja sitten yhden ruudun sivuun valitusta suunnasta.
021         * Maalipiste siis voi olla jokin kahdeksasta ruudusta ratsun
022         * ympärillä. Ratsu voi myös hyppiä muiden nappuloiden ylitse.
023         *
024         * @return <code>true</code>, jos ehdotettu siirto on sallittu ratsulle
025         * @param peli <code>Shakkipeli</code>, jossa siirto halutaan tehdä
026         * @param vaakaAlku lähtökohdan vaakakoordinaatti [A-Ha-h]
027         * @param pystyAlku lähtökohdan pystykoordinaatti [1-8]
028         * @param vaakaLoppu maaliruudun vaakakoordinaatti [A-Ha-h]
029         * @param pystyLoppu maaliruudun pystykoordinaatti [1-8]
030         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaAlku == c) && (1 <= pystyAlku <= 8) && EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaLoppu == c) && (1 <= pystyLoppu <= 8)</code>
031         */
032        public boolean onLaillinen(Shakkipeli peli, char vaakaAlku, int pystyAlku,
033                                                                                                char vaakaLoppu, int pystyLoppu) {
034                // maaliruudussa ei saa olla omaa nappulaa
035                if (peli.onMiehitetty(vaakaLoppu, pystyLoppu)) {
036                        if (peli.annaNappula(vaakaLoppu, pystyLoppu).onValkoinen() == onValkoinen()) {
037                                return false;
038                        }
039                }
040                if ((Math.abs(vaakaLoppu - vaakaAlku) == 2 && Math.abs(pystyLoppu - pystyAlku) == 1) ||
041                        (Math.abs(pystyLoppu - pystyAlku) == 2 && Math.abs(vaakaLoppu - vaakaAlku) == 1)) {
042                        return true;
043                }
044                return false;
045        }
046        
047        /**
048         * Vertaa tätä <code>Ratsu</code>:a toiseen olioon.
049         * Palauttaa <code>true</code>, jos toinen
050         * on samanvärinen <code>Ratsu</code>
051         * -nappula, tätä ja toista on tai ei ole
052         * siirretty meneillää olevassa pelissä ja
053         * molemmilla on sama merkkiesitys.
054         *
055         * @param o tähän verrattava olio
056         * @return <code>true</code>, jos tämä <code>Ratsu</code> on samanlainen kuin verrukki
057         */
058        public boolean equals(Object o) {
059                if (o == this) return true;
060                if (o instanceof Ratsu) {
061                        return super.equals(o);
062                }
063                return false;
064        }       
065}

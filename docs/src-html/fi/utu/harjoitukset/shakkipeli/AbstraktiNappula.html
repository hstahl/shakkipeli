001package fi.utu.harjoitukset.shakkipeli;
002/**
003 * Shakkipelin abstrakti pelinappulaluokka, jossa annetaan toteutus niille
004 * <code>Nappula</code>:n ominaisuuksille, jotka ovat yhteisiä kaikille
005 * rajapinnan määritelmän mukaisille shakkinappuloille.
006 */
007public abstract class AbstraktiNappula implements Nappula {
008
009        /** Merkkiesitys nappulalle pelilaudalla. */
010        protected final String MERKKIESITYS;
011        /** Onko tämä nappula valkoisen? */
012        protected final boolean onValkoinen;
013        /** jotkut siirrot ovat sallittuja vain ensimmäistä kertaa siirrettäessä */
014        protected boolean ensimmainenSiirto;
015
016        /**
017         * <code>AbstraktiNappula</code>:n aliluokista kutsuttava konstruktori. Asettaa
018         * kaikille nappuloille yhteiset luokkamuuttujat.
019         *
020         * @param merkkiesitys <code>Nappula</code>:n merkkiesitys, joka on jokin <code>Nappula.MerkkiesitysNappulalle</code> literaaliluokan vakioista
021         * @param onValkoinen Onko tämä nappula valkoisen pelaajan?
022         * @.pre <code>valkoinen != null && musta != null</code>
023         * @.post <code>onValkoinen() == onValkoinen && toString().equals(merkkiesitys) && onSiirretty() == false</code>
024         */
025        protected AbstraktiNappula(MerkkiesitysNappulalle merkkiesitys, boolean onValkoinen) {
026                MERKKIESITYS = onValkoinen ? merkkiesitys.valkoinen() : merkkiesitys.musta();
027                this.onValkoinen = onValkoinen;
028                ensimmainenSiirto = true;
029        }
030        
031        /**
032         * Kertoo, onko nappula valkoinen
033         *
034         * @return <code>true</code>, jos nappula on valkoisen pelaajan.
035         */
036        public boolean onValkoinen() {
037                return onValkoinen;
038        }
039        
040        /** 
041         * Havannoi, onko nappulaa sirretty. Joitakin siirtoja
042         * voi tehdä vain ensimmäisellä kerralla siirrettäessä
043         * nappulaa.
044         * 
045         * @return <code>true</code>, jos nappulaa on sirretty
046         */
047        public boolean onSiirretty() {
048                return !ensimmainenSiirto;
049        }
050        
051        /**
052         * Abstrakti totuusarvometodi nappulalle sallittujen siirtojen
053         * tarkastamista varten.
054         *
055         * @return <code>true</code>, jos ehdotettu siirto on sallittu tälle nappulalle
056         * @param peli <code>Shakkipeli</code>, jossa siirto halutaan tehdä
057         * @param vaakaAlku lähtökohdan vaakakoordinaatti [A-Ha-h]
058         * @param pystyAlku lähtökohdan pystykoordinaatti [1-8]
059         * @param vaakaLoppu maaliruudun vaakakoordinaatti [A-Ha-h]
060         * @param pystyLoppu maaliruudun pystykoordinaatti [1-8]
061         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaAlku == c) && (1 <= pystyAlku <= 8) && EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaLoppu == c) && (1 <= pystyLoppu <= 8)</code>
062         */
063        public abstract boolean onLaillinen(Shakkipeli peli, char vaakaAlku, int pystyAlku,
064                                                                                                                 char vaakaLoppu, int pystyLoppu);
065        
066        /**
067         * Vertaa oliota tahan abstraktiin nappulaan.
068         * Palauttaa <code>true</code>, jos olio on
069         * myös abstrakti shakkinappula, se on
070         * samanvärinen kuin tämä, sillä on sama
071         * merkkiesitys ja molempia on tai ei ole
072         * siirretty meneillään olevassa shakkipelissä.
073         *
074         * @param o tähän verrattava olio
075         * @return <code>true</code>, jos verrattava olio on samanlainen nappula
076         */
077        public boolean equals(Object o) {
078                if (!(o instanceof AbstraktiNappula)) return false;
079                AbstraktiNappula temp = (AbstraktiNappula) o;
080                return (temp.onValkoinen() == onValkoinen())
081                        && (temp.onSiirretty() == onSiirretty())
082                        && (temp.toString().equals(toString()));
083        }
084        
085        /**
086         * Shakkinappulan merkkijonoesitys
087         *
088         * @return <code>Nappula</code>:n esitys String:nä
089         */
090        public String toString() {
091                return MERKKIESITYS;
092        }
093}

007/**
008 * Konsolissa pelattava shakkipeli, joka ei tiedä
009 * shakin säännöistä muuta, kuin nappuloiden
010 * sallitut liikkeet. Tekoälyvastustajaa ei ole.
011 * <p>
012 * Shakkipeli koostuu pelilaudasta, joka on 8x8
013 * ruudukko, ja mustista ja valkoisista pelinappuloista.
014 * Pelilaudan ruudut numeroidaan kirjain- ja
015 * numeroyhdistelmällä siten, että, pelaajien
016 * kuvakulmasta, sarakkeet ovat nimetty A-H ja rivit 1-8.
017 * <p>
018 * Pelaajat eivät voi tehdä laittomia siirtoja
019 * lukuunottamatta pelin päättymistilanteita, joissa
020 * kuningas on uhattuna. Siispä pelin päättyminen
021 * ja kuninkaan siirtäminen vaaratilanteeseen ovat
022 * pelaajien itsensä vastuulla. Shakin säännöt
023 * löytyvät vaikkapa
024 * <a href=http://fi.wikipedia.org/wiki/Shakki>wikipediasta</a>.
025 * <p>
026 * Shakkia pelataan komentorivillä vuorotellen
027 * syöttäen koordinaatteja ensin valiten nappula,
028 * jota halutaan siirtää ja sitten kohderuutu. Peli
029 * kertoo, jos siirto onnistui. Jos siirto oli laiton,
030 * peli kehottaa yrittämään uudestaan.
031 * <p>
032 * Pelissä on myös kello, joka näyttää kummankin
033 * pelaajan siirtoihin käyttämän ajan aina vuoron
034 * päätyttyä.
035 * <p>
036 * <h5>Käyttö</h5>
037 * Shakkipeliä pelataan komentoriviltä. Aja tämän
038 * luokan <code>main()</code> -metodi JVM:llä komennolla
039 * <code>java fi.utu.harjoitukset.shakkipeli.Shakkipeli</code>
040 * projektin juurikansiosta.
041 * Ohjelma tulostaa ruudulle käyttöohjeita ajettaessa.
042 * <p>
043 * <h5>Pelin kulku</h5>
044 * Shakin pelaaminen alkaa, kun käyttäjä syöttää komennon
045 * uuden pelin aloittamiseksi tai pelin lataamiseksi
046 * tiedostosta. Pelilauta tulostuu näkyviin ja pelaajat
047 * voivat alkaa syöttää koordinaatteja siirtoja varten.
048 * Koordinaatit syötetään yksi koordinaattipari kerrallaan
049 * välilöynnillä erotettuna. Ensin valitaan siirrettävä
050 * nappula, jonka onnistuessa ohjelma viestittää konsoliin.
051 * Seuraavaksi valitaan kohderuutu. Jos siirto onnistui,
052 * vuoro vaihtuu vastustajalle. Muussa tapauksessa, peli
053 * viestii miksi siirto ei onnistunut ja pelaajan tulee
054 * aloittaa siirtäminen uudestaan nappulan valitsemisesta.
055 * <p>
056 * Peli päättyy, kun pelaajat itse huomaavat jomman kumman
057 * kuninkaan olevan pelastamattomassa pinteessä. Shakkipeli
058 * ei itse viestitä tai keskeytä peliä shakkimatin sattuessa,
059 * eikä myöskään estä pelaajia siirtämästä kuningasta shakkiin,
060 * mikä on säännöissä kielletty.
061 */

119        /**
120         * Shakkipelin päälohko, josta löytyy pelin logiikka.
121         * Tätä kutsutaan, kun halutaan aloittaa peli.
122         */
123        public void aloita() {
124                // siirtoja varten käytetään Koordinaatit olion sisältämiä lukupareja
125                Koordinaatit<Character,Integer> koord = new Koordinaatit<Character,Integer>(new Character(' '),new Integer(-1));            
126                // Siirtoja varten. Jos jotain on pielessä, pitäisi alkaa ropista IndexOutOfBoundsException:eja.
127                char vaakaAlku = ' ', vaakaLoppu = ' ';
128                int pystyAlku = -1, pystyLoppu = -1;
129                /* Siirtojen koordinaatit annetaan kahdessa askeleessa */ 
130                int askel = 0;
131                
132                System.out.print("Kirjoita konsoliin \"chcp 1257\" saadaksesi ääkköset näkyviin\n");
133                System.out.print("\n");
134                System.out.print("-----SHAKKIPELI-----\n");
135                System.out.print("\n");
136                System.out.print("KOMENNOT:\n");
137                System.out.print("Aloita uusi peli: uusi\n");
138                System.out.print("Lataa tallennettu peli tiedostosta: load <tiedosto>\n");
139                System.out.print("Tallenna nykyinen peli tiedostoon: save <tiedosto>\n");
140                System.out.print("Kuinka pelata: ohje\n");
141                System.out.print("Sulje peli: quit\n");
142
143                // main loop
144                while (!lopeta) {
145                        String komento = pyydaSyote();
146                        try { parsiKomento(komento); } // kokeillaan onko syöte komento
147                        catch (TunnistamatonKomentoPoikkeus e) { // jos ei...
148                                try { // kokeillaan onko syöte koordinaatit
149                                        koord = parsiKoordinaatit(komento);
150                                        pelaaVuoro = true;
151                                }
152                                catch (TunnistamatonKomentoPoikkeus f) {
153                                        System.out.println(e); continue; // ei komento eikä koordinaatti
154                                }
155                        } // tästä eteenpäin vain, jos komento tunnistettiin
156                        
157                        if (pelaaVuoro && !peliAlkanut) {
158                                System.out.println("Aloita uusi peli tai lataa tiedostosta ennen kuin syötät koordinaatteja!");
159                                pelaaVuoro = false;
160                        }
161                        
162                        // Varsinaisen shakkipelin pelivuoro
163                        if (pelaaVuoro && peliAlkanut) {
164                                askel++;
165                                
166                                // Valitaan siirrettävä nappula
167                                if (askel == 1) {
168                                        vaakaAlku = koord.annaEka().charValue();
169                                        pystyAlku = koord.annaToka().intValue();
170                                        if (onMiehitetty(vaakaAlku,pystyAlku)) {
171                                                if (annaNappula(vaakaAlku,pystyAlku).onValkoinen() == valkoisenVuoro) {
172                                                        System.out.println("Nappula valittu!");
173                                                }
174                                                else {
175                                                        System.out.println("Et voi siirtää vastustajan nappulaa!");
176                                                        askel = 0;
177                                                }
178                                        }
179                                        else {
180                                                System.out.println("Ruudussa ei ole nappulaa!");
181                                                askel = 0;
182                                        }
183                                }
184                                
185                                // Valitaan mihin ruutuun nappulaa siirretään
186                                else if (askel == 2) {
187                                        vaakaLoppu = koord.annaEka().charValue();
188                                        pystyLoppu = koord.annaToka().intValue();
189                                        if (annaNappula(vaakaAlku,pystyAlku).onLaillinen(this, vaakaAlku, pystyAlku, vaakaLoppu, pystyLoppu)) {
190                                                if (onMiehitetty(vaakaLoppu,pystyLoppu)) {
191                                                        syoNappula(vaakaLoppu,pystyLoppu);
192                                                }
193                                                siirraNappula(vaakaAlku, pystyAlku, vaakaLoppu, pystyLoppu);
194                                                
195                                                // sotilasnappuloille on vähän poikkeuksia
196                                                if (annaNappula(vaakaLoppu, pystyLoppu).getClass().equals(Sotilas.class)) {
197                                                        // Kun sotilas pääsee pelilaudan päätyyn, se ylennetään
198                                                        if (pystyLoppu == 1 || pystyLoppu == 8) {
199                                                                System.out.println("Sotilas ylennetään. " +
200                                                                                                   "Valitse haluamasi upseeriarvo: l, r, t tai d " +
201                                                                                                   "(lähetti, ratsu, torni tai kuningatar).");
202                                                                boolean onnistui = false;
203                                                                while (!onnistui) {
204                                                                        try {
205                                                                                komento = pyydaSyote();
206                                                                                ylennaSotilas(vaakaLoppu, pystyLoppu, komento);
207                                                                                onnistui = true;
208                                                                        }
209                                                                        catch (TunnistamatonKomentoPoikkeus e) {
210                                                                                System.out.println("Virheellinen syöte! Kokeile uudestaan!");
211                                                                        }
212                                                                }
213                                                                System.out.println("Ylennys onnistui!");
214                                                        }
215                                                        // jos liikuttiin kaksi ruutua sotilaalla, merkitään ohestalyötäväksi
216                                                        else if (Math.abs(pystyLoppu - pystyAlku) == 2) {
217                                                                ohestalyotava = new Koordinaatit<Character,Integer>(new Character(vaakaLoppu), new Integer(pystyLoppu));
218                                                        }
219                                                        // Syödään ohestalöynnille haavoittuvainen nappula
220                                                        else if (annaOhestalyotava() != null) {
221                                                                if (pystyAlku == annaOhestalyotava().annaToka() &&
222                                                                        vaakaLoppu == annaOhestalyotava().annaEka()) {
223                                                                        syoNappula(vaakaLoppu, pystyAlku);
224                                                                }
225                                                                ohestalyotava = null;
226                                                        }
227                                                }
228                                                // jos ei liikutettu sotilasta, yksikään nappula ei ole enää ohestalyötävä
229                                                else ohestalyotava = null;
230                                                
231                                                // Jos siirto oli linnoitus, siirretään myös tornia
232                                                if (annaNappula(vaakaLoppu, pystyLoppu).getClass().equals(Kuningas.class)) {                                            
233                                                        if (pystyAlku == pystyLoppu && Math.abs(vaakaLoppu - vaakaAlku) == 2) {
234                                                                int kumpiPuoli = 0;
235                                                                if (vaakaAlku < vaakaLoppu) kumpiPuoli = 1;
236                                                                else if (vaakaAlku > vaakaLoppu) kumpiPuoli = -2;
237                                                                if (onMiehitetty((char)(vaakaLoppu + kumpiPuoli), pystyLoppu) && kumpiPuoli != 0) {
238                                                                        if (annaNappula((char)(vaakaLoppu + kumpiPuoli), pystyLoppu).getClass().equals(Torni.class)) {
239                                                                                if (!annaNappula((char)(vaakaLoppu + kumpiPuoli), pystyLoppu).onSiirretty()) {
240                                                                                        siirraNappula((char)(vaakaLoppu + kumpiPuoli), pystyLoppu, (kumpiPuoli > 0) ? (char)(vaakaLoppu - 1): (char)(vaakaLoppu + 1), pystyAlku);
241                                                                                }
242                                                                                else System.err.println("Tornia on jo siirretty. Virhe Kuningas luokassa.");
243                                                                        }
244                                                                        else System.err.println("Kulmassa oleva nappula ei ole Torni. Virhe Kuningas luokassa.");
245                                                                }
246                                                                else System.err.println("Kulmassa ei ole nappulaa. Virhe Kuningas luokassa.");
247                                                        }
248                                                }                                       
249                                                askel = 0;
250                                                System.out.println("Siirto onnistui!");
251                                                valkoisenVuoro = valkoisenVuoro ? false : true;
252                                                System.out.println(this); // siirron onnistuttua pitää päivittää näkymä
253                                        }
254                                        else {
255                                                System.out.println("Siirto ei sallittu! Yritä uudestaan!");
256                                                askel = 0;
257                                        }
258                                }
259                                else { // tänne ei jouduta
260                                        System.err.println("Fatal Error! Askeleen arvo ei sallittu: " + askel);
261                                        System.exit(-1);
262                                }
263                                
264                                // Tämä vuoro on ohi. Seuraavan kerran tähän lohkoon
265                                // tullaan, kun syötteeksi saadaan koordinaatit.
266                                pelaaVuoro = false;
267                                
268                                // lopuksi päivitetään vuorossa olleen kello
269                                if (valkoisenVuoro) valkoisenKello += System.currentTimeMillis() - annaAika();
270                                else mustanKello += System.currentTimeMillis() - annaAika();
271                                // aloitetaan seuraavan vuoron ajanotto
272                                otaAika();
273                        }       
274                }
275        }

336        /**
337         * Asettaa uuden nappulan pelilaudalle. Jos annetussa ruudussa
338         * on ennestään nappula, vanha nappula korvautuu uudella.
339         *
340         * @see #asetaSiirrettyNappula(char,int,char,boolean)
341         * @param vaaka kohderuudun vaakakoordinaatti [A-Ha-h]
342         * @param pysty kohderuudun pystykoordinaatti [1-8]
343         * @param tyyppi Haluttu nappulan tyyppi. On oltava jokin seuraavista: s, l, r, t, d, k.
344         * @param onValkoinen Onko asetettava nappula valkoisen pelaajan?
345         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; c == Character.toUpperCase(vaaka)) && (1 <= pysty <= 8) && EXISTS(char c : new char[]{'s','l','r','t','d','k'}; tyyppi == c)</code>
346         * @.post <code>onMiehitetty(vaaka, pysty) == true</code>
347         */
348        private void asetaNappula(char vaaka, int pysty, char tyyppi, boolean onValkoinen) {
349                switch (tyyppi) {
350                        case 's':
351                                pelilauta.asetaNappula(new Sotilas(onValkoinen), vaaka, pysty);
352                                break;
353                        case 'l':
354                                pelilauta.asetaNappula(new Lahetti(onValkoinen), vaaka, pysty);
355                                break;
356                        case 'r':
357                                pelilauta.asetaNappula(new Ratsu(onValkoinen), vaaka, pysty);
358                                break;
359                        case 't':
360                                pelilauta.asetaNappula(new Torni(onValkoinen), vaaka, pysty);
361                                break;
362                        case 'd':
363                                pelilauta.asetaNappula(new Kuningatar(onValkoinen), vaaka, pysty);
364                                break;
365                        case 'k':
366                                pelilauta.asetaNappula(new Kuningas(onValkoinen), vaaka, pysty);
367                                break;
368                        default:
369                }
370        }

372        /**
373         * Asettaa uuden siirretyn nappulan pelilaudalle.
374         * Jos annetussa ruudussa on ennestään nappula,
375         * vanha nappula korvautuu uudella. Käytetään
376         * ladatessa peliä tiedostosta.
377         *
378         * @see #load(String)
379         * @see #asetaNappula(char,int,char,boolean)
380         * @param vaaka kohderuudun vaakakoordinaatti [A-Ha-h]
381         * @param pysty kohderuudun pystykoordinaatti [1-8]
382         * @param tyyppi Haluttu nappulan tyyppi. On oltava jokin seuraavista: s, t, k.
383         * @param onValkoinen Onko asetettava nappula valkoisen pelaajan?
384         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; c == Character.toUpperCase(vaaka)) && (1 <= pysty <= 8) && EXISTS(char c : new char[]{'s','t','k'}; tyyppi == c)</code>
385         * @.post <code>onMiehitetty(vaaka, pysty) && annaNappula(vaaka, pysty).onSiirretty()</code>
386         */     
387        private void asetaSiirrettyNappula(char vaaka, int pysty, char tyyppi, boolean onValkoinen) {
388                switch (tyyppi) {
389                        case 's':
390                                pelilauta.asetaNappula(new Sotilas(onValkoinen, true), vaaka, pysty);
391                                break;
392                        case 't':
393                                pelilauta.asetaNappula(new Torni(onValkoinen, true), vaaka, pysty);
394                                break;
395                        case 'k':
396                                pelilauta.asetaNappula(new Kuningas(onValkoinen, true), vaaka, pysty);
397                                break;
398                        default:
399                }       
400        }

402        /**
403         * Poistaa koordinaateissa olevan nappulan pelilaudalta
404         * ja siirtää sen syötyjen nappuloiden listaan.
405         * 
406         * @param vaaka syötävän nappulan vaakakoordinaatti [A-Ha-h]
407         * @param pysty syötävän nappulan pystykoordinaatti [1-8]
408         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; c == Character.toUpperCase(vaaka)) && (1 <= pysty <= 8) && onMiehitetty(vaaka, pysty)</code>
409         * @.post <code>onMiehitetty(vaaka, pysty) == false && vainajat.contains(OLD.annaNappula(vaaka, pysty))</code>
410         */
411        private void syoNappula(char vaaka, int pysty) {
412                vainajat.add(pelilauta.annaNappula(vaaka, pysty));
413                pelilauta.poistaNappula(vaaka, pysty);
414        }
415        
416        /**
417         * Siirtää nappulan alkuruudusta kohderuutuun. Syödessä
418         * vastustajan nappulaa on tarkoitus käyttää syoNappula()
419         * -metodia ensin. Siirron laillisuus tulee tarkastaa
420         * ennen käyttöä pelin aikana.
421         *
422         * @see #syoNappula(char,int)
423         * @param vaakaAlku siirrettävän nappulan vaakakoordinaatti [A-Ha-h]
424         * @param pystyAlku siirrettävän nappulan pystykoordinaatti [1-8]
425         * @param vaakaLoppu kohderuudun vaakakoordinaatti [A-Ha-h]
426         * @param pystyLoppu kohderuudun pystykoordinaatti [1-8]
427         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; c == Character.toUpperCase(vaakaAlku)) && (1 <= pystyAlku <= 8) && EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; c == Character.toUpperCase(vaakaLoppu)) && (1 <= pystyLoppu <= 8) && onMiehitetty(vaakaAlku, pystyAlku)</code>
428         * @.post <code>OLD.annaNappula(vaakaAlku, pystyAlku) == annaNappula(vaakaLoppu, pystyLoppu) && onMiehitetty(vaakaAlku, pystyAlku) == false</code>
429         */
430        private void siirraNappula(char vaakaAlku, int pystyAlku, char vaakaLoppu, int pystyLoppu) {
431                pelilauta.asetaNappula(pelilauta.annaNappula(vaakaAlku, pystyAlku), vaakaLoppu, pystyLoppu);
432                pelilauta.poistaNappula(vaakaAlku, pystyAlku);
433        }

435        /**
436         * Asettaa pelilaudalle nappulat uutta peliä varten
437         * aloituspaikoilleen ja resetoi muuttujat.
438         */
439        private void alustaUusiPeli() {
440                // tyhjennetään pelilauta
441                pelilauta = new Pelilauta<Nappula>();
442

563        /**
564         * Yrittää lukea komentomerkkijonosta oikein muotoillun
565         * koordinaattiparin ja palauttaa sen <code>Koordinaatit</code>
566         * -oliona, joka sisältää koordinaattiparin. Ensimmäinen on
567         * vaaka- ja toinen pystykoordinaatti. Parametrin
568         * <code>komento</code> tulee olla muotoa "[A-Ha-h] [1-8]".
569         *
570         * @return luetut koordinaatit <code>Koordinaatit</code> -oliona
571         * @param komento parsittava komento
572         * @.pre <code>komento != null</code>
573         * @.post <code>(RESULT.annaEka().toString().matches("[A-Ha-h]")) && (1 <= RESULT.annaToka() <= 8)</code>
574         * @throws TunnistamatonKomentoPoikkeus jos komento on väärin muotoiltu tai koordinaatit eivät ole validit
575         */
576        private Koordinaatit<Character,Integer> parsiKoordinaatit(String komento) throws TunnistamatonKomentoPoikkeus {
577                Character eka;
578                Integer toka;
579                Scanner lukija = new Scanner(komento);
580                String luettu = "";
581                if (lukija.hasNext()) luettu = lukija.next();
582                else throw new TunnistamatonKomentoPoikkeus(komento);
583                if (luettu.matches("[A-Ha-h]")) eka = new Character(luettu.charAt(0));
584                else throw new TunnistamatonKomentoPoikkeus(komento);
585                if (lukija.hasNext()) luettu = lukija.next();
586                else throw new TunnistamatonKomentoPoikkeus(komento);
587                try { toka = Integer.parseInt(luettu); }
588                catch (NumberFormatException e) { throw new TunnistamatonKomentoPoikkeus(komento); }
589                if (toka < 1 || toka > 8) throw new TunnistamatonKomentoPoikkeus(komento);
590                return new Koordinaatit<Character,Integer>(eka,toka);
591        }

625        private void save(String tiedostonimi) {
626                try {
627                        FileOutputStream out = new FileOutputStream(new File(tiedostonimi));
628                        // aluksi kirjoitetaan peliruudun tilanne
629                        for (char vaaka : new char[]{'A','B','C','D','E','F','G','H'}) {
630                                for (int pysty = 1; pysty <= 8; pysty++) {

633                                         * -ominaisuutta, eli kaikki 128:aa suuremmat arvot
634                                         * ovat valkoisia nappuloita.
635                                         * 1xxx xxxx valkoinen
636                                         * 0xxx xxxx musta
637                                         *
638                                         * Kolme vähiten merkitsevää
639                                         * bittiä määräävät nappulan tyypin seuraavasti:
640                                         * 001 sotilas
641                                         * 010 lähetti
642                                         * 011 Ratsu
643                                         * 100 Torni
644                                         * 101 Kuningatar
645                                         * 110 Kuningas
646                                         *
647                                         * Neljäs bitti merkitsee joidenkin nappuloiden
648                                         * onSiirretty() ominaisuutta.
649                                         * 1xxx onSiirretty()
650                                         * 0xxx !onSiirretty() 
651                                         */
652                                        int val = 0;
653                                        if (onMiehitetty(vaaka, pysty)) {
654                                                if (annaNappula(vaaka, pysty).onValkoinen()) {
655                                                        val += 128;
656                                                }
657                                                switch (annaNappula(vaaka, pysty).getClass().getSimpleName()) {
658                                                        case "Sotilas":
659                                                                val += 1;
660                                                                break;
661                                                        case "Lahetti":
662                                                                val += 2;
663                                                                break;
664                                                        case "Ratsu":
665                                                                val += 3;
666                                                                break;
667                                                        case "Torni":

669                                                                break;
670                                                        case "Kuningatar":
671                                                                val += 5;
672                                                                break;
673                                                        case "Kuningas":
674                                                                val += 6;
675                                                                break;
676                                                        default:
677                                                }
678                                                if (annaNappula(vaaka, pysty).onSiirretty()) val += 8;
679                                        }
680                                        // tyhjään ruutuun kirjoitetaan nolla
681                                        else val = 0;
682                                        out.write(val);
683                                }
684                        }
685                        // käytetään välimerkkinä tavua 0xff
686                        out.write(0xff);
687                        // seuraavaksi syötyjen nappuloiden lista
688                        for (Nappula n : vainajat) {
689                                int val = 0;
690                                if (n.onValkoinen()) val += 128;
691                                switch (n.getClass().getSimpleName()) {
692                                        case "Sotilas":
693                                                val += 1;
694                                                break;
695                                        case "Lahetti":
696                                                val += 2;
697                                                break;
698                                        case "Ratsu":
699                                                val += 3;
700                                                break;
701                                        case "Torni":
702                                                val += 4;
703                                                break;
704                                        case "Kuningatar":
705                                                val += 5;
706                                                break;
707                                        case "Kuningas":
708                                                val += 6;
709                                                break;
710                                        default:                
711                                }
712                                out.write(val);
713                        }
714                        out.write(0xff); // vainajien lista päättyi
715                        
716                        // seuraavaksi ohestalyötävän koordinaatit
717                        if (ohestalyotava != null) {
718                                // ByteBuffer luokan array() palauttaa Byte taulukon, joka voidaan antaa sellaisenaan kirjoitettavaksi
719                                out.write(ByteBuffer.allocate(2).putChar(ohestalyotava.annaEka().charValue()).array());
720                                out.write(ByteBuffer.allocate(4).putInt(ohestalyotava.annaToka().intValue()).array());
721                                out.write(0xff);
722                        }
723                        else out.write(0xff); // käytetään merkkaamaan ohestalyöntiä null:ksi

728                        
729                        // lopuksi vielä talletetaan kumman vuoroon jäätiin
730                        out.write(valkoisenVuoro ? 1 : 0);
731                        out.flush();
732                        out.close();
733                }
734                catch (FileNotFoundException fnfe) {
735                        System.err.println("Kohde on kansio tai muokkaaminen ei sallittu: " + fnfe);
736                }
737                catch (NullPointerException nulle) {
738                        // tätä ei pitäisi tapahtua, koska tiedostonimen täytyisi olla null

747        /**
748         * Lataa pelitilanteen tiedostosta.
749         *
750         * @see #save(String)
751         * @see #asetaSiirrettyNappula(char,int,char,boolean)
752         * @param tiedostonimi tallennetun pelin tiedostostonimi
753         * @.pre <code>tiedostonimi != null</code>
754         */
755        private void load(String tiedostonimi) {
756                try {
757                        FileInputStream in = new FileInputStream(new File(tiedostonimi));
758                        LinkedList<Integer> luetut = new LinkedList<Integer>();
759                        // koska javan tavut ovat kahden komplementteja, käytetään AND operaatiota unsigned arvon saamiseksi
760                        while (in.available() > 0) {
761                                luetut.add(new Integer(in.read()) & 0xff);
762                        }
763                        int luettu = 0;
764                        // ensiksi luetaan pelilaudan nappulat
765                        pelilauta = new Pelilauta<Nappula>();
766                        boolean asetaValkoinen = false;
767                        boolean asetaSiirretty = false;
768                        char vaaka = 'A';
769                        int pysty = 1;
770                        luettu = luetut.removeFirst();
771                        while (luettu != 0xff) {
772                                if (luettu > 128) {
773                                        asetaValkoinen = true;
774                                        luettu -= 128;
775                                }
776                                else asetaValkoinen = false;
777                                if (luettu > 8) {
778                                        asetaSiirretty = true;
779                                        luettu -= 8;
780                                }
781                                else asetaSiirretty = false;
782                                switch (luettu) {
783                                        case 0: break; // ruudussa ei ole nappulaa, eikä haluta ajaa default -lohkoa
784                                        case 1:
785                                                if (asetaSiirretty) asetaSiirrettyNappula(vaaka, pysty, 's', asetaValkoinen);
786                                                else asetaNappula(vaaka, pysty, 's', asetaValkoinen);
787                                                break;
788                                        case 2:
789                                                asetaNappula(vaaka, pysty, 'l', asetaValkoinen);
790                                                break;
791                                        case 3:
792                                                asetaNappula(vaaka, pysty, 'r', asetaValkoinen);
793                                                break;
794                                        case 4:
795                                                if (asetaSiirretty) asetaSiirrettyNappula(vaaka, pysty, 't', asetaValkoinen);
796                                                else asetaNappula(vaaka, pysty, 't', asetaValkoinen);
797                                                break;
798                                        case 5:
799                                                asetaNappula(vaaka, pysty, 'd', asetaValkoinen);
800                                                break;
801                                        case 6:
802                                                if (asetaSiirretty) asetaSiirrettyNappula(vaaka, pysty, 'k', asetaValkoinen);
803                                                else asetaNappula(vaaka, pysty, 'k', asetaValkoinen);
804                                                break;
805                                        default:
806                                                System.out.println("Virhe lukiessa tiedostosta pelilautaa: " + luettu);
807                                }
808                                if (pysty < 8) pysty++;
809                                else {pysty = 1; vaaka++;}
810                                luettu = luetut.removeFirst();
811                        } // while
812
813                        // seuraavaksi luetaan syötyjen nappuloiden lista
814                        vainajat = new LinkedList<Nappula>();
815                        luettu = luetut.removeFirst();
816                        while (luettu != 0xff) {
817                                if (luettu > 128) {
818                                        asetaValkoinen = true;
819                                        luettu -= 128;
820                                }
821                                else asetaValkoinen = false;
822                                switch (luettu) {
823                                        case 1:
824                                                vainajat.add(new Sotilas(asetaValkoinen));
825                                                break;
826                                        case 2:
827                                                vainajat.add(new Lahetti(asetaValkoinen));
828                                                break;
829                                        case 3:
830                                                vainajat.add(new Ratsu(asetaValkoinen));
831                                                break;
832                                        case 4:
833                                                vainajat.add(new Torni(asetaValkoinen));
834                                                break;
835                                        case 5:
836                                                vainajat.add(new Kuningatar(asetaValkoinen));
837                                                break;
838                                        case 6:
839                                                vainajat.add(new Kuningas(asetaValkoinen));
840                                                break;
841                                        default:
842                                                System.out.println("Virhe lukiessa tiedostosta syötyjä nappuloita: " + luettu);                         
843                                }
844                                luettu = luetut.removeFirst();
845                        }
846                        // seuraavaksi ohestalyötävän koordinaatit
847                        luettu = luetut.removeFirst();
848                        if (luettu != 0xff) {
849                                int val = 0;
850                                // char on 2 tavua
851                                for (int i = 0; i < 2; i++) {
852                                        val = val << 8;
853                                        val = val + luettu;
854                                        luettu = luetut.removeFirst();
855                                }
856                                char c = (char)val;
857                                val = 0;
858                                // int on 4 tavua
859                                for (int i = 0; i < 4; i++) {
860                                        val = val << 8;
861                                        val = val + luettu;
862                                        luettu = luetut.removeFirst();
863                                }
864                                ohestalyotava = new Koordinaatit<Character,Integer>(new Character(c), new Integer(val));
865                        }
866                        else ohestalyotava = null;
867                        
868                        valkoisenKello = 0;
869                        mustanKello = 0;
870                        // muutetaan seuraavat 8 tavua yhdeksi long:ksi
871                        for (int i = 0; i < 8; i++) {
872                                valkoisenKello <<= 8;
873                                valkoisenKello += luetut.removeFirst();
874                        }
875                        
876                        // muutetaan seuraavat 8 tavua yhdeksi long:ksi
877                        for (int i = 0; i < 8; i++) {
878                                mustanKello <<= 8;
879                                mustanKello += luetut.removeFirst();
880                        }
881                        
882                        valkoisenVuoro = (luetut.removeFirst() == 0) ? false : true;
883                        
884                        in.close();
885                        
886                        // peli voi jatkua
887                        System.out.println(this);
888                        otaAika();
889                        peliAlkanut = true;
890                }
891                catch (FileNotFoundException fnfe) {
892                        System.err.println("Tiedostoa ei löytynyt");
893                }
894                catch (IOException IOe) {
895                        System.err.println("Tiedostoa ei voitu lukea: " + IOe);
896                }
897                catch (NoSuchElementException noee) {
898                        System.err.println("EOF ladattaessa peliä tiedostosta");
899                }

902        /**
903         * Palauttaa merkkiesityksen shakkilaudasta
904         * ja pelitilanteesta. Windows konsolin
905         * rajoitteiden takia käytetään nappuloiden
906         * merkintöinä iso- ja pikkukirjaimia.
907         * <p>
908         * Miltä uuden pelin pitäisi näyttää:
909         * <p><code>&nbsp;
910         *        A B C D E F G H<br />
911         *      8|t|r|l|d|k|l|r|t|<br />
912         *      7|s|s|s|s|s|s|s|s|  0:00<br />
913         *      6| | | | | | | | |<br />
914         *      5| | | | | | | | |<br />
915         *      4| | | | | | | | |<br />
916         *      3| | | | | | | | |<br />
917         *      2|S|S|S|S|S|S|S|S|  0:00<br />
918         *      1|T|R|L|D|K|L|R|T|</code>
919         *
920         * @return merkkijonopiirros pelitilanteesta
921         */
922        public String toString() {
923                String pelinakyma = "";
924                
925                // peliruudukon tulostus nappuloineen
926                pelinakyma += "\n  A B C D E F G H\n";
927                int rivi = 1;
928                for (rivi = 8; rivi >= 1; rivi--) {
929                        pelinakyma += (rivi) + "|";
930                        for (char sarake : new char[]{'A','B','C','D','E','F','G','H'}) {
931                                pelinakyma += onMiehitetty(sarake,rivi) ? annaNappula(sarake,rivi).toString() : " ";
932                                pelinakyma += "|";
933                        }
934                        // Kellot riveille 2 ja 7
935                        if (rivi == 7) pelinakyma += "  " + sekuntikello(mustanKello);
936                        if (rivi == 2) pelinakyma += "  " + sekuntikello(valkoisenKello);
937
938                        pelinakyma += "\n";
939                }
940                
941                pelinakyma += "\n";
942                
943                // Syodyt nappulat
944                pelinakyma += "Syödyt: ";
945                for (Nappula n : vainajat) {
946                        pelinakyma += n.toString() + " ";
947                }
948                pelinakyma += "\n\n";
949                // kumman vuoro?
950                pelinakyma += valkoisenVuoro ? "Valkoisen siirto" : "Mustan siirto";            
951                return pelinakyma;
952        }

961        public static String sekuntikello(long millisek) {

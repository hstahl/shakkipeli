001package fi.utu.harjoitukset.shakkipeli;
002import java.lang.Math;
003/**
004 * Sotilas- shakkinappula
005 */
006public class Sotilas extends AbstraktiNappula {
007        
008        /**
009         * Uutta sotilasta luodessa kutsuttava konstruktori
010         *
011         * @param onValkoinen Halutaanko luoda valkoinen nappula?
012         * @.post <code>this.onValkoinen() == onValkoinen && this.onSiirretty() == false</code>
013         */
014        public Sotilas(boolean onValkoinen) {
015                super(MerkkiesitysNappulalle.SOTILAS, onValkoinen);
016        }
017        
018        /**
019         * Konstruktori tiedostosta ladattua peliä varten, jossa
020         * sotilasta on saatettu jo siirtää
021         * 
022         * @param onValkoinen Halutaanko luoda valkoinen nappula?
023         * @param onSiirretty Onko nappulaa siirretty ladattavassa pelissä?
024         * @.post <code>this.onValkoinen() == onValkoinen && this.onSiirretty() == onSiirretty</code>
025         */
026        public Sotilas(boolean  onValkoinen, boolean onSiirretty) {
027                super(MerkkiesitysNappulalle.SOTILAS, onValkoinen);
028                ensimmainenSiirto = !onSiirretty;
029        }
030        
031        /**
032         * Sotilas liikkuu suoraan eteenpäin yhden ruudun, jos edessä ei ole vastustajan
033         * nappulaa, ja etuviistoon syödessä nappulan. Poikkeuksia on kaksi: ensimmäisen
034         * kerran siirrettäessä sotilasta voidaan siirtää kaksi ruutua eteen, jos
035         * kummassakaan ruudussa ei ole toista nappulaa, ja jos vastustajan sotilas
036         * liikkuu kaksi ruutua eteen tullen sotilaan viereen, se voidaan syödä nk.
037         * ohestalyönti -liikkeellä, jolloin tavalliseen tapaan sirrytään etuviistoon
038         * vastustajan sotilaan taakse ja poistetaan se pelilaudalta. Jos sotilas kulkee
039         * pelilaudan toiseen päähän saakka, se ylennetään upseeriksi (ratsu, lähetti,
040         * torni tai kuningatar), mutta tämä tarkistus tehdään <code>Shakkipeli</code> -luokassa.
041         *
042         * @return      <code>true</code>, jos ehdotettu siirto on sallittu sotilaalle
043         * @param peli <code>Shakkipeli</code>, jossa siirto halutaan tehdä
044         * @param vaakaAlku lähtökohdan vaakakoordinaatti [A-Ha-h]
045         * @param pystyAlku lähtökohdan pystykoordinaatti [1-8]
046         * @param vaakaLoppu maaliruudun vaakakoordinaatti [A-Ha-h]
047         * @param pystyLoppu maaliruudun pystykoordinaatti [1-8]
048         * @.pre <code>EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaAlku == c) && (1 <= pystyAlku <= 8) && EXISTS(char c : new char[]{'A','B','C','D','E','F','G','H'}; vaakaLoppu == c) && (1 <= pystyLoppu <= 8)</code>
049         * @.post <code>(RESULT && onSiirretty()) || !RESULT</code>
050         */
051        public boolean onLaillinen(Shakkipeli peli, char vaakaAlku, int pystyAlku,
052                                                                                                char vaakaLoppu, int pystyLoppu) {
053                boolean result = false;
054                if (onValkoinen()) { // valkoisille nappuloille
055                        if (vaakaAlku == vaakaLoppu) {
056                                if (pystyLoppu - pystyAlku == 2) {
057                                        // yritetään siirtää kaksi ruutua eteenpäin
058                                        if (ensimmainenSiirto &&
059                                                !(peli.onMiehitetty(vaakaLoppu, pystyLoppu - 1) ||
060                                                peli.onMiehitetty(vaakaLoppu, pystyLoppu))) {
061                                                result = true;
062                                        }
063                                }
064                                else if (pystyLoppu - pystyAlku == 1) {
065                                        // yritetään siirtää yksi ruutu eteenpäin
066                                        if (!(peli.onMiehitetty(vaakaLoppu, pystyLoppu))) {
067                                                result = true;
068                                        }
069                                }
070                        }
071                        else if (Math.abs(vaakaLoppu - vaakaAlku) == 1 && pystyLoppu - pystyAlku == 1) {
072                                // yritetään syödä nappula
073                                if (peli.onMiehitetty(vaakaLoppu, pystyLoppu)) {
074                                        if (!(peli.annaNappula(vaakaLoppu, pystyLoppu).onValkoinen())) {
075                                                result = true;
076                                        }
077                                }
078                                // jos loppuruutu ei ollut miehitetty, tarkastetaan vielä ohestalyönti
079                                else if (peli.annaOhestalyotava() != null) {
080                                        if (peli.annaOhestalyotava().annaEka() == vaakaLoppu &&
081                                                peli.annaOhestalyotava().annaToka() == pystyAlku) {
082                                                result = true;
083                                        }
084                                }
085                        }
086                }
087                else { // mustille nappuloille
088                        if (vaakaAlku == vaakaLoppu) {
089                                if (pystyLoppu - pystyAlku == -2) {
090                                        // yritetään siirtää kaksi ruutua eteenpäin
091                                        if (ensimmainenSiirto &&
092                                                !(peli.onMiehitetty(vaakaLoppu, pystyLoppu + 1) ||
093                                                peli.onMiehitetty(vaakaLoppu, pystyLoppu))) {
094                                                result = true;
095                                        }
096                                }
097                                else if (pystyLoppu - pystyAlku == -1) {
098                                        // yritetään siirtää yksi ruutu eteenpäin
099                                        if (!(peli.onMiehitetty(vaakaLoppu, pystyLoppu))) {
100                                                result = true;
101                                        }
102                                }
103                        }
104                        else if (Math.abs(vaakaLoppu - vaakaAlku) == 1 && pystyLoppu - pystyAlku == -1) {
105                                // yritetään syödä nappula
106                                if (peli.onMiehitetty(vaakaLoppu, pystyLoppu)) {
107                                        if (peli.annaNappula(vaakaLoppu, pystyLoppu).onValkoinen()) {
108                                                result = true;
109                                        }
110                                }
111                                // jos loppuruutu ei ollut miehitetty, tarkastetaan vielä ohestalyönti
112                                else if (peli.annaOhestalyotava() != null) {
113                                        if (peli.annaOhestalyotava().annaEka() == vaakaLoppu &&
114                                                peli.annaOhestalyotava().annaToka() == pystyAlku) {
115                                                result = true;
116                                        }
117                                }
118                        }
119                }
120                if (result) ensimmainenSiirto = false;
121                return result;
122        }
123        
124        /**
125         * Vertaa tätä <code>Sotilas</code>:ta toiseen olioon.
126         * Palauttaa <code>true</code>, jos toinen
127         * on samanvärinen <code>Sotilas</code>
128         * -nappula, tätä ja toista on tai ei ole
129         * siirretty meneillää olevassa pelissä ja
130         * molemmilla on sama merkkiesitys.
131         *
132         * @param o tähän verrattava olio
133         * @return <code>true</code>, jos tämä <code>Sotilas</code> on samanlainen kuin verrukki
134         */
135        public boolean equals(Object o) {
136                if (o == this) return true;
137                if (o instanceof Sotilas) {
138                        return super.equals(o);
139                }
140                return false;
141        }
142}
